    def generate_figure(self, stack: ImageStack) -> Figure | None:
        geometry = stack.geometry

        if geometry is None:
            return None

        # Apply dark styling
        plt.style.use("dark_background")
        figure: Figure = show_geometry(geometry, show=False).figure
        figure.set_facecolor("black")

        return figure
