class OpHistoryCopyDialogModel:
    def __init__(self, images):

[docs]
    def apply_ops(self, ops: Iterable[ImageOperation], copy: bool):
        if copy:
            self.images = self.images.copy()

        to_apply = ops_to_partials(ops)
        for op in to_apply:
            op(self.images)
        return self.images
